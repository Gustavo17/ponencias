\subsection{Columnas}

\begin{frame}{Las columnas}
	Nos permite dividir en varias columnas nuestros contenidos:
	\begin{columns}[c]
		\column{0.5\textwidth}
			\begin{block}{Izquierda}
				\begin{itemize}
					\item Un listado
					\item Dentro de un bloque
					\item Dentro de una columna
				\end{itemize}
			\end{block}
		\column{0.5\textwidth}
			\begin{block}{Derecha}
				$\left|{1\over N}\sum_{n=1}^N \gamma(u_n)-{1\over 2\pi}\int_0^{2\pi}\gamma(t){\rm d}t\right|$
				\begin{itemize}
					\item Una formula y un listado.
					\item Dentro de un bloque.
					\item Dentro de una columna.
				\end{itemize}
			\end{block}
	\end{columns}
\end{frame}

\begin{frame}{Opciones de las columnas}
	\begin{itemize}
		\item b: Alinea las columnas por la ultima linea.
		\item c: Alinea las columnas por el centro.
		\item t: Alinea las columnas por la primera linea (por defecto).
		\item T: Alinea las columnas por su punto mas arriba.
		\item totalwidth: Tamaño maximo del conjunto de columnas.
		\item onlytextwidth igual que totalwidth=\\textwidth.
	\end{itemize}
\end{frame}

    \end{block}

\section{Efectos}
\subsection{Overlays}
\begin{frame}[containsverbatim]{Overlays}
	\begin{itemize}
		\item Separan nuestra presentacion en capas.
		\item Estas capas puede ocultarse o mostrarse.
		\item Se les puede establecer un orden de aparicion.
		\item Se suelen especificar con la expresion \verb+<n>+.
		\item Ej: \verb+\item<1>+, \verb+\begin{block}{Bloque}<1-3>\end{block}+, \verb+\item<3->+.
		\item Se pueden establecer pausas con \verb+\pause+.
		\item Se pueden ocultar o omitir los contenidos \verb+\only<1> != \item<1>+
	\end{itemize}
\end{frame}

\begin{frame}{Ejemplos de Overlays}
	\begin{block}{Bloque1}<1>Este es el bloque 1\end{block}
	\begin{block}{Bloque2}<2>Este es el bloque 2\end{block}
	\only<3>{
		\begin{block}{Bloque 3}Este es el bloque 3\end{block}
	}
	\begin{block}{Bloque4}<1->\color<2-4>[rgb]{1,0,0}Este es el bloque 4\end{block}
\end{frame}
