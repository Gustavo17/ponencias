  \frame{\titlepage}
  
  \begin{frame}
    \frametitle{¿Qué es el GUL?}

    \begin{block}{El GUL}
    \begin{itemize}
      \item Es una asociación de la Universidad Carlos III de Madrid.
      \item Es el Grupo de Usuarios de Linux de la Universidad Carlos III de Madrid.
      \item Dedicada a promover el uso del software libre.
      \item Consta de mas de 300 socios de todo el mundo.
      \item Ha colaborado en la organización de congresos de primera linea en
            el ámbito del software como son Hispalinux y Javahispano.
    \end{itemize}
    \end{block}
  \end{frame}

  \end{frame}

    \frametitle{Grupos}
    \begin{block}{Tres clases de grupos}
    \begin{itemize}
      \item Junta: Grupo encargado de la toma de decisiones.
      \item Grupos técnicos: Grupos encargados de llevar a buen puerto los proyectos del GUL.
      \item Grupos de interés: Grupos con intereses comunes.
    \end{itemize}
    \end{block}
  \end{frame}
