  \section{Introducción}

  \begin{frame}{Introducción}
    \begin{center}Introducción\end{center}
  \end{frame}
  
  \begin{frame}{¿Que significa funcional?}
    \begin{itemize}
      \item Programación con funciones (matemáticas)
      \item Funciones puras (mismas entradas, mismas salidas).
      \item Inmutabilidad.
      \item Ausencia de estado.
    \end{itemize}
  \end{frame}

  \begin{frame}{¿Por qué?}
    \begin{itemize}
      \item Paralelización.
      \item Facilitar el testing.
      \item Reusabilidad.
      \item Depuración más fácil.
      \item Estado muy controlado.
    \end{itemize}
  \end{frame}
  
  \begin{frame}{Típicas estrategias funcionales}
    \begin{itemize}
      \item Combinación y composición de funciones pequeñas.
      \item Datos + funciones transformadoras.
      \item Aplicación de transformaciones mediante orden superior.
      \item Uso de funciones inline.
      \item Acotado de efectos laterales.
      \item Tendencia al uso de funciones puras.
    \end{itemize}
  \end{frame}

  \begin{frame}{¿Qué necesito?}
    \begin{itemize}
      \item Funciones como ciudadanos de primera (son un objeto más).
    \end{itemize}
  \end{frame}

  \begin{frame}{¿Es python un lenguaje funcional?}
    \begin{itemize}
      \item No.
      \item Es un lenguaje multi-paradigma.
      \item Soporta algunas características funcionales.
      \item Permite hacer programación funcional.
      \item Carece de características avanzadas presentes en lenguajes funcionales.
    \end{itemize}
  \end{frame}

  \begin{frame}{¿Que me dan los lenguajes funcionales?}
    \begin{itemize}
      \item Estructuras inmutables eficientes.
      \item Funciones de orden superior.
      \item Pattern matching.
      \item TCO: Tail call optimization.
      \item Aplicación parcial y currificación.
      \item Control de efectos laterales.
      \item Funciones lambda.
      \item Evaluación perezosa.
      \item Composición de funciones.
    \end{itemize}
  \end{frame}

  \begin{frame}{¿Que me da python?}
    \begin{itemize}
      \item Evaluación perezosa (Limitada).
      \item Aplicación parcial.
      \item Funciones lambda.
      \item Funciones de orden superior.
    \end{itemize}
  \end{frame}

  \begin{frame}{¿Que me da fn.py?}
    \begin{itemize}
      \item Estructuras inmutables eficientes (En desarrollo).
      \item TCO: Tail call optimization.
      \item Aplicación parcial y currificación.
      \item Composición de funciones.
      \item Funciones lambda (Al estilo de Scala).
    \end{itemize}
  \end{frame}

  \begin{frame}[containsverbatim]{Funcional vs. Imperativo}
    \begin{block}{Imperativo}
       \begin{verbatim}
x = sum(1, 2)
y = sum(x, 3)
z = prod(y, 4)
print(z)
       \end{verbatim}
    \end{block}
    \begin{block}{Funcional}
       \begin{verbatim}
print(prod(sum(sum(1,2),3), 4))
       \end{verbatim}
    \end{block}
  \end{frame}

  \begin{frame}[containsverbatim]{Funcional vs. Imperativo}
    \begin{block}{Funcional con composición y aplicación parcial}
       \begin{verbatim}
func = F(sum, 1, 2) >> F(sum, 3) >> F(prod, 4) >> print
func()
       \end{verbatim}
    \end{block}
    \begin{block}{Funcional con currificación}
       \begin{verbatim}
prod4 = prod(4)
sum3 = sum(3)
sum2 = sum(2)
print(prod4(sum3(sum2(1))))
       \end{verbatim}
    \end{block}
  \end{frame}

  \begin{frame}[containsverbatim]{TCO}
    \begin{block}{Recursión normal}
      \begin{verbatim}
def fact(n):
    if n == 0: return 1
        return n * fact(n-1)
      \end{verbatim}
    \end{block}

    \begin{block}{TCO}
      \begin{verbatim}
from fn import recur

@recur.tco
def fact(n, acc=1):
    if n == 0: return False, acc
    return True, (n-1, acc*n)
      \end{verbatim}
    \end{block}
  \end{frame}

  \begin{frame}[containsverbatim]{Aplicación parcial}
    \begin{block}{Aplicación parcial}
      \begin{verbatim}
>>> from fn import F
>>> from operator import add
>>> add2 = F(add, 2)
>>> add2(3)
5
      \end{verbatim}
    \end{block}
  \end{frame}

  \begin{frame}[containsverbatim]{Currificación}
    \begin{block}{Currificación}
      \begin{verbatim}
>>> from fn.func import curried
>>> curry_add = curried(lambda x, y: x + y)
>>> curry_add(2)(3)
5
>>> @curried
... def curried_add(x, y):
...     return x + y
...
>>> curried_add(2)(3)
5
      \end{verbatim}
    \end{block}
  \end{frame}
  
  \begin{frame}[containsverbatim]{Composición de funciones}
    \begin{block}{Composición normal}
      \begin{verbatim}
>>> myfunc = lambda x: duplicate(add2(x))
>>> myfunc(3)
10
      \end{verbatim}
    \end{block}

    \begin{block}{Composición al estilo fn.py}
      \begin{verbatim}
>>> myfunc = F(duplicate) << add2
>>> myfunc(3)
10
>>> myfunc = F(add2) >> duplicate
>>> myfunc(3)
10
      \end{verbatim}
    \end{block}
  \end{frame}

  \begin{frame}[containsverbatim]{Funciones lambda al estilo scala}
