  \section{Introducción}
  
  \begin{frame}{¿Que significa funcional?}
    \begin{block}{¿Que significa funcional?}
      \begin{itemize}
        \item Programación con funciones (matematicas)
        \item Funciones puras (mismas entradas, mismas salidas).
        \item Inmutabilidad.
        \item Ausencia de estado.
      \end{itemize}
    \end{block}
  \end{frame}

  \begin{frame}{¿Por qué?}
    \begin{block}{¿Por qué?}
      \begin{itemize}
        \item Paralelización.
        \item Facilitar el testing.
        \item Reusabilidad.
        \item Depuración más fácil.
      \end{itemize}
    \end{block}
  \end{frame}
  
  \begin{frame}{¿Qué necesito?}
    \begin{block}{¿Qué necesito?}
      \begin{itemize}
        \item Funciones como ciudadanos de primera (son un objeto más).
      \end{itemize}
    \end{block}
  \end{frame}

  \begin{frame}{¿Es python funcional?}
    \begin{block}{¿Es python funcional?}
      \begin{itemize}
        \item No, es un lenguaje multi-paradigma.
        \item Soporta algunas caracteristicas funcionales.
        \item Permite hacer programación funcional.
        \item Carece de caracteristicas avanzadas presentes en lenguajes funcionales.
      \end{itemize}
    \end{block}
  \end{frame}

  \begin{frame}{Tipicas estrategias funcionales}
    \begin{block}{Tipicas estrategias funcionales}
      \begin{itemize}
        \item Combinación y composición de funciones pequeñas.
        \item Datos + funciones transformadoras.
        \item Aplicación de transformaciones mediante orden superior.
        \item Uso de funciones inline.
        \item Acotado de efectos laterales.
        \item Tendencia al uso de funciones puras.
      \end{itemize}
    \end{block}
  \end{frame}

  \begin{frame}{¿Que me dan los lenguajes funcionales?}
    \begin{block}{¿Que me dan los lenguajes funcionales?}
      \begin{itemize}
        \item Estructuras inmutables eficientes.
        \item Funciones de orden superior.
        \item Pattern matching.
        \item TCO: Tail call optimization.
        \item Aplicación parcial y currificación.
        \item Control de efectos laterales.
        \item Funciones lambda.
        \item Evaluación peresoza.
        \item Composición de funciones.
      \end{itemize}
    \end{block}
  \end{frame}

  \begin{frame}{¿Que me da python?}
    \begin{block}{¿Que me da python?}
      \begin{itemize}
        \item Evaluación peresoza (Limitada).
        \item Aplicación parcial.
        \item Funciones lambda.
        \item Funciones de orden superior.
      \end{itemize}
    \end{block}
  \end{frame}

  \begin{frame}{¿Que me da fn.py?}
    \begin{block}{¿Que me da fn.py?}
      \begin{itemize}
        \item Estructuras inmutables eficientes (En desarrollo).
        \item TCO: Tail call optimization.
        \item Aplicación parcial y currificación.
        \item Composición de funciones.
        \item Funciones lambda (Al estilo de Scala).
      \end{itemize}
    \end{block}
  \end{frame}

  \begin{frame}[containsverbatim]{Funcional vs. Imperativo}
    \begin{block}{Imperativo}
       \begin{verbatim}
x = sum(1, 2)
y = sum(x, 3)
z = prod(y, 4)
print(z)
       \end{verbatim}
    \end{block}
    \begin{block}{Funcional}
       \begin{verbatim}
print(prod(sum(sum(1,2),3), 4))
       \end{verbatim}
    \end{block}
  \end{frame}

  \begin{frame}[containsverbatim]{Funcional vs. Imperativo}
    \begin{block}{Funcional con composición y aplicacion parcial}
       \begin{verbatim}
func = F(sum, 1, 2) >> F(sum, 3) >> F(prod, 4) >> print
func()
       \end{verbatim}
    \end{block}
    \begin{block}{Funcional con currificación}
       \begin{verbatim}
prod4 = prod(4)
sum3 = sum(3)
sum2 = sum(2)
print(prod4(sum3(sum2(1))))
       \end{verbatim}
    \end{block}
  \end{frame}

    \begin{block}{Funciones de orden superior}
       \begin{verbatim}
>>> list(map(lambda x: x**2, [1, 2, 3]))
[1, 4, 9]
>>> list(filter(lambda x: x > 1, [1, 2, 3]))

>>> sorted([2, 1, 3], key=lambda x: x)
[1, 2, 3]
>>> sorted([1, 2, 3], key=lambda x: -x)
[3, 2, 1]
>>> from functools import reduce
>>> reduce(lambda x, y: x + y, [1, 2, 3])
6
>>> from functools import lru_cache
>>> cached_sum = lru_cache()(lambda x: sum(range(x)))
>>> cached_sum(4)
6

  \begin{frame}[containsverbatim]{TCO}
    \begin{block}{Normal recursion}
      \begin{verbatim}
def fact(n):
    if n == 0: return 1
        return n * fact(n-1)
      \end{verbatim}
    \end{block}

    \begin{block}{TCO}
      \begin{verbatim}
from fn import recur

@recur.tco
def fact(n, acc=1):
    if n == 0: return False, acc
    return True, (n-1, acc*n)
      \end{verbatim}
    \end{block}
  \end{frame}

  \begin{frame}[containsverbatim]{Applicación parcial}
    \begin{block}{Applicación parcial}
      \begin{verbatim}
>>> from fn import F
>>> from operator import add
>>> add2 = F(add, 2)
>>> add2(3)
5
      \end{verbatim}
    \end{block}
  \end{frame}

  \begin{frame}[containsverbatim]{Currificacion}
    \begin{block}{Currificacion}
      \begin{verbatim}
>>> from fn.func import curried
>>> curry_add = curried(lambda x, y: x + y)
>>> curry_add(2)(3)
5
>>> @curried
... def curried_add(x, y):
...     return x + y
...
>>> curried_add(2)(3)
5
      \end{verbatim}
    \end{block}
  \end{frame}
  
  \begin{frame}[containsverbatim]{Composición de funciones}
    \begin{block}{Composición normal}
      \begin{verbatim}
>>> myfunc = lambda x: duplicate(add2(x))
>>> myfunc(3)
10
      \end{verbatim}
    \end{block}

    \begin{block}{Composición al estilo fn.py}
      \begin{verbatim}
>>> myfunc = F(duplicate) << add2
>>> myfunc(3)
10
>>> myfunc = F(add2) >> duplicate
>>> myfunc(3)
10
      \end{verbatim}
    \end{block}
  \end{frame}

  \begin{frame}[containsverbatim]{Funciones lambda al estilo scala}
    \begin{block}{Funciones lambda al estilo scala}

>>> from fn import _
>>> (_ + _)(2, 3)

>>> list(map(_ + 2, [1, 2, 3]))
[3, 4, 5]
      \end{verbatim}
    \end{block}
  \end{frame}
